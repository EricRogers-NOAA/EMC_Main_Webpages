109  ENDIF
110

113 
114  IF((ityp.EQ.2).OR.(ityp.EQ.3)) THEN

118  missj = .NOT.(herej)

120  IF(herei.AND.missj) THEN

122  nmrg = nmrg+1
123  ELSEIF(herei.AND.herej.AND..NOT.samei) THEN

125  ENDIF
