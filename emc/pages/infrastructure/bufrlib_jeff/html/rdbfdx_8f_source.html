84 C-----------------------------------------------------------------------
85 C-----------------------------------------------------------------------
86 
87  CALL dxinit(lun,0)
88 
89  ict = 0
90  done = .false.
91 
92 C Read a complete dictionary table from LUNIT, as a set of one or
93 C more DX dictionary messages.
94 
95  DO WHILE ( .NOT. done )
96  CALL rdmsgw( lunit, mgwa, ier )
97  IF ( ier .EQ. -1 ) THEN
98 
99 C Don't abort for an end-of-file condition, since it may be
100 C possible for a file to end with dictionary messages.
101 C Instead, backspace the file pointer and let the calling
102 C routine diagnose the end-of-file condition and deal with
103 C it as it sees fit.
104 
105  CALL backbufr(lun)
106  done = .true.
107  ELSE IF ( ier .EQ. -2 ) THEN
108  goto 900
109  ELSE IF ( idxmsg(mgwa) .NE. 1 ) THEN
110 
111 C This is a non-DX dictionary message. Assume we've reached
112 C the end of the dictionary table, and backspace LUNIT so that
113 C the next read (e.g. in the calling routine) will get this
114 C same message.
115 
116  CALL backbufr(lun)
117  done = .true.
118  ELSE IF ( iupbs3(mgwa,'NSUB') .EQ. 0 ) THEN
119 
120 C This is a DX dictionary message, but it doesn't contain any
121 C actual dictionary information. Assume we've reached the end
122 C of the dictionary table.
123 
124  done = .true.
125  ELSE
126 
127 C Store this message into MODULE TABABD.
128 
129  ict = ict + 1
130  CALL stbfdx(lun,mgwa)
131  ENDIF
132  ENDDO
133 
134  IF ( iprt .GE. 2 ) THEN
135  CALL errwrt('+++++++++++++++++++++++++++++++++++++++++++++++++')
136  WRITE ( unit=errstr, fmt='(A,I3,A)' )
137  . 'BUFRLIB: RDBFDX - STORED NEW DX TABLE CONSISTING OF (',
138  . ict, ') MESSAGES;'
139  CALL errwrt(errstr)
140  errstr = 'WILL APPLY THIS TABLE TO ALL SUBSEQUENT DATA IN '//
141  . 'FILE UNTIL NEXT DX TABLE IS FOUND'
142  CALL errwrt(errstr)
143  CALL errwrt('+++++++++++++++++++++++++++++++++++++++++++++++++')
144  CALL errwrt(' ')
145  ENDIF
146 
147  CALL makestab
148 
149  RETURN
150  900 CALL bort('BUFRLIB: RDBFDX - ERROR READING A BUFR DICTIONARY '//
151  . 'MESSAGE')
152  END
