4 C> THIS FUNCTION UNPACKS AND RETURNS A SPECIFIED INTEGER VALUE
5 C> FROM SECTION 0 OR SECTION 1 OF THE BUFR MESSAGE STORED IN ARRAY
6 C> MBAY. IT WILL WORK ON ANY MESSAGE ENCODED USING BUFR EDITION 2, 3
7 C> OR 4. THE START OF THE BUFR MESSAGE (I.E. THE STRING "BUFR") MUST
8 C> BE ALIGNED ON THE FIRST FOUR BYTES OF MBAY, AND THE VALUE TO BE
9 C> UNPACKED IS SPECIFIED VIA THE MNEMONIC S01MNEM, AS EXPLAINED IN

97 C-----------------------------------------------------------------------
98 C This statement function checks whether its input value contains
99 C a valid century value.
100 
101  ok4cent(ival) = ((ival.GE.19).AND.(ival.LE.21))
102 C-----------------------------------------------------------------------
103 
104 C Call subroutine WRDLEN to initialize some important information
105 C about the local machine, just in case subroutine OPENBF hasn't
106 C been called yet.
107 
108  CALL wrdlen
109 
110 C Handle some simple requests that do not depend on the BUFR
111 C edition number.
112 
113  IF(s01mnem.EQ.'LENM') THEN
114  iupbs01 = iupb(mbay,5,24)
115  RETURN
116  ENDIF
117 
118  len0 = 8
119  IF(s01mnem.EQ.'LEN0') THEN
120  iupbs01 = len0
121  RETURN
122  ENDIF
123 
124 C Get the BUFR edition number.
125 
126  iben = iupb(mbay,8,8)
127  IF(s01mnem.EQ.'BEN') THEN
128  iupbs01 = iben
129  RETURN
130  ENDIF
131 
132 C Use the BUFR edition number to handle any other requests.
133 
134  CALL gets1loc(s01mnem,iben,isbyt,iwid,iret)
135  IF(iret.EQ.0) THEN
136  iupbs01 = iupb(mbay,len0+isbyt,iwid)
137  IF(s01mnem.EQ.'CENT') THEN
138 
139 C Test whether the returned value was a valid
140 C century value.
141 
142  IF(.NOT.ok4cent(iupbs01)) iupbs01 = -1
143  ENDIF
144  ELSE IF( (s01mnem.EQ.'YEAR') .AND. (iben.LT.4) ) THEN
145 
146 C Calculate the 4-digit year.
147 
148  iyoc = iupb(mbay,21,8)
149  icen = iupb(mbay,26,8)
150 
151 C Does ICEN contain a valid century value?
152 
153  IF(ok4cent(icen)) THEN
154 
155 C YES, so use it to calculate the 4-digit year. Note that,
156 C by international convention, the year 2000 was the 100th
157 C year of the 20th century, and the year 2001 was the 1st
158 C year of the 21st century
159 
160  iupbs01 = (icen-1)*100 + iyoc
161  ELSE
162 
163 C NO, so use a windowing technique to determine the
164 C 4-digit year from the year of the century.
165 
166  iupbs01 = i4dy(mod(iyoc,100)*1000000)/10**6
167  ENDIF
168  ELSE
169  iupbs01 = -1
170  ENDIF
171 
172  RETURN
173  END
