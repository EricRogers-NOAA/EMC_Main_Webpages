109  SUBROUTINE readmg(LUNXX,SUBSET,JDATE,IRET)
110 
111  USE moda_msgcwd
112  USE moda_sc3bfr
113  USE moda_bitbuf
114 
115  include 'bufrlib.inc'
116 
117  COMMON /quiet / iprt
118 
119  CHARACTER*128 errstr
120  CHARACTER*8 subset
121 
122 C-----------------------------------------------------------------------
123 C-----------------------------------------------------------------------
124 
125  iret = 0
126  lunit = abs(lunxx)
127 
128 C CHECK THE FILE STATUS
129 C ---------------------
130 
131  CALL status(lunit,lun,il,im)
132  IF(il.EQ.0) goto 900
133  IF(il.GT.0) goto 901
134  CALL wtstat(lunit,lun,il,1)
135 
136 C READ A MESSAGE INTO THE INTERNAL MESSAGE BUFFER
137 C -----------------------------------------------
138 
139 1 CALL rdmsgw(lunit,mbay(1,lun),ier)
140  IF(ier.EQ.-1) goto 200
141 
142 C PARSE THE MESSAGE SECTION CONTENTS
143 C ----------------------------------
144 
145  IF(isc3(lun).NE.0) CALL reads3(lun)
146  CALL cktaba(lun,subset,jdate,iret)
147 
148 C LOOK FOR A DICTIONARY MESSAGE
149 C -----------------------------
150 
151  IF(idxmsg(mbay(1,lun)).NE.1) RETURN
152 
153 C This is an internal dictionary message that was
154 C generated by the BUFRLIB archive library software.
155 
156  IF(isc3(lun).NE.0) RETURN
157 
158 C Section 3 decoding isn't being used, so backspace the
159 C file pointer and then use subroutine RDBFDX to read in
160 C all such dictionary messages (they should be stored
161 C consecutively!) and reset the internal tables.
162 
163  CALL backbufr(lun)
164  CALL rdbfdx(lunit,lun)
165 
166  IF(iprt.GE.1) THEN
167  CALL errwrt('++++++++++++++BUFR ARCHIVE LIBRARY+++++++++++++++++')
168  errstr = 'BUFRLIB: READMG - INTERNAL DICTIONARY MESSAGE READ;'//
169  .' ACCOUNT FOR IT THEN READ IN NEXT MESSAGE WITHOUT RETURNING'
170  CALL errwrt(errstr)
171  CALL errwrt('++++++++++++++BUFR ARCHIVE LIBRARY+++++++++++++++++')
172  CALL errwrt(' ')
173  ENDIF
174 
175 C Now go read another message.
176 
177  goto 1
178 
179 C EOF ON ATTEMPTED READ
180 C ---------------------
181 
182 200 CALL wtstat(lunit,lun,il,0)
183  inode(lun) = 0
184  idate(lun) = 0
185  subset = ' '
186  jdate = 0
187  iret = -1
188  RETURN
189

191 C -----
192

194  . ' BE OPEN FOR INPUT')
195 901 CALL bort('BUFRLIB: READMG - INPUT BUFR FILE IS OPEN FOR OUTPUT'//
196  . ', IT MUST BE OPEN FOR INPUT')
